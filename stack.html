Stack Implementation

A stack is an abstract data type (ADT) commonly used in programming languages. It follows the LIFO (Last-In-First-Out) principle, similar to a real-world stack of cards or plates.

Stack Operations

  • push: Insert an element onto the stack
  • pop: Remove and access the top element from the stack
  • peek: Get the top element without removing it
  • isFull: Check if the stack is full
  • isEmpty: Check if the stack is empty

Basic Stack Implementation


    #include <stdio.h>
    #include <stdbool.h>

    #define MAXSIZE 100

    int stack[MAXSIZE];
    int top = -1;

    void push(int data) {
      if (!isFull()) {
        top++;
        stack[top] = data;
      } else {
        printf("Could not insert data, Stack is full.\n");
      }
    }

    int pop() {
      if (!isEmpty()) {
        int data = stack[top];
        top--;
        return data;
      } else {
        printf("Could not retrieve data, Stack is empty.\n");
        return -1;
      }
    }

    int peek() {
      if (!isEmpty()) {
        return stack[top];
      } else {
        printf("Stack is empty.\n");
        return -1;
      }
    }

    bool isFull() {
      if (top == MAXSIZE - 1)
        return true;
      else
        return false;
    }

    bool isEmpty() {
      if (top == -1)
        return true;
      else
        return false;
    }

    int main() {
      push(5);
      push(10);
      push(15);
      printf("Peek: %d\n", peek());
      printf("Pop: %d\n", pop());
      printf("Pop: %d\n", pop());
      printf("Peek: %d\n", peek());
      return 0;
    }
